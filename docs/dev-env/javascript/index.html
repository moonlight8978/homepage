.prettierrc.yml
singleQuote: true

Babel#

yarn add -D @babel/core @babel/preset-env @babel/plugin-transform-runtime
babel.config.js
module.exports = {
presets: [
["@babel/preset-env", { targets: "defaults" }],
],
plugins: ["@babel/plugin-transform-runtime"],
};

Typescript#

yarn add -D @babel/preset-typescript typescript
yarn tsc --init
babel.config.js
module.exports = {
presets: [
"@babel/preset-typescript",
],
};
tsconfig.json
{
"compilerOptions": {
"lib": ["ESNext"],
"allowJs": true,
}
}
.eslintrc.yml
rules:
no-use-before-define: 'off'
'@typescript-eslint/no-use-before-define': error
no-unused-vars: 'off'
'@typescript-eslint/no-unused-vars':
- error
- args: none
varsIgnorePattern: '^_'
no-useless-constructor: 'off'
'@typescript-eslint/no-useless-constructor':
- error
no-shadow: 'off'
'@typescript-eslint/no-shadow':
- error

Module alias#

yarn add --dev babel-plugin-module-resolver
babel.config.js
module.exports = {
plugins: [
[
"module-resolver",
{
root: ["."],
extensions: [".ts", ".tsx", ".js", ".jsx", "json"],
alias: {
"@src": "./src",
},
},
],
],
};

ESLint#

yarn add -D eslint-import-resolver-typescript
.eslintrc.yml
settings:
import/resolver:
typescript: {}
rules:
import/order:
- error
- pathGroups:
- pattern: '@src/**'
group: parent
groups:
- builtin
- external
- parent
- sibling
- index
newlines-between: always

Typescript#

tsconfig.json
{
"compilerOptions": {
"baseUrl": "./",
"paths": {
"@src/*": ["./src/*"]
}
}
}

Jest#

jest.config.ts
export default {
moduleNameMapper: {
"^@src/(.*)$": "<rootDir>/your/folder/$1",
},
};
